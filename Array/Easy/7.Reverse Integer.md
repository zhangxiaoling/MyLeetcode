class Solution(object):
    def reverse(self, x):
        """
        :type x: int
        :rtype: int
        """
        result = 0
        sign=(x>0)-(x<0)
        x=x*sign
        while x:
            result = result * 10 + x % 10
            x /= 10

        return sign*result*(result<2**31)
