# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def reverseList(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        if head==None:
            return None
        elif head.next==None:
            return head
        else:
            new_head=self.reverseList(head.next)
            head.next.next=head
            head.next=None
            return new_head

class Solution(object):
    def reverseList(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        if head==None:
            return None
        else:
            p=head
            q=head.next
            head.next=None
            while q:
                r=q.next
                q.next=p
                p=q
                q=r
            head=p
            return head
